CREATE TABLE HISTORY_PERSONAL(
    PERSONAL_ID INT(11) NOT NULL,
	PERSONAL_CROSSED_EYE VARCHAR(1) NOT NULL,
    PERSONAL_LAZY_EYE VARCHAR(1) NOT NULL,
    PERSONAL_DROOPING_EYELID VARCHAR(1) NOT NULL,
    PERSONAL_PROMINENT_EYE VARCHAR(1) NOT NULL,
    PERSONAL_GLAUCOMA VARCHAR(1) NOT NULL,
    PERSONAL_RETINAL_DISEASE VARCHAR(1) NOT NULL,
    PERSONAL_CATARACTS VARCHAR(1) NOT NULL,
    PERSONAL_EYE_INFECTIONS VARCHAR(1) NOT NULL,
    PERSONAL_EYE_INJURY VARCHAR(1) NOT NULL,
    PRIMARY KEY(PERSONAL_ID))
	
	
	
	
	
	
	
	
	
	
	
	
	
	CREATE TABLE PATIENT(
    PATIENT_ID INT(11) NOT NULL AUTO_INCREMENT,
    PATIENT_NAME_LAST VARCHAR(52) NOT NULL,
    PATIENT_NAME_FIRST VARCHAR(102) NOT NULL,
    PATIENT_NAME_MIDDLE VARCHAR(52),
    PATIENT_BDAY DATE NOT NULL,
    PATIENT_SEX VARCHAR(1) NOT NULL,
    PATIENT_HOME VARCHAR(102) NOT NULL,
    PATIENT_CITY INT(11) NOT NULL,
    PATIENT_COUNTRY_ID INT(11) NOT NULL,
    PATIENT_OCCUPATION VARCHAR(52) NOT NULL,
    PATIENT_COMPANY_NAME VARCHAR(52) NOT NULL,
    PATIENT_COMPANY_NUMBER INT(11) NOT NULL,
    PATIENT_SPORTS INT(11) NOT NULL,
    PATIENT_EMAIL1 VARCHAR(52) NOT NULL,
    PATIENT_EMAIL2 VARCHAR(52),
    PATIENT_FACEBOOK VARCHAR(52),
    PATIENT_LANDLINE INT(11),
    PATIENT_CELL INT(11),
    PATIENT_ALLERGIES VARCHAR(300),
    PATIENT_MEDICATIONS VARCHAR(300),
    PATIENT_INJURIES_SURGERIES_HOSPITALIZATION VARCHAR(500),
    PATIENT_HISTORY_PERSONAL_ID INT(11) NOT NULL,
    PATIENT_HISTORY_FAMILY_ID INT(11) NOT NULL,
    PATIENT_PERSONAL_EXPERIENCE INT(11) NOT NULL,
    PATIENT_REFERRED_BY VARCHAR(102),
    PATIENT_HEALTH_CARD INT(11),
    PATIENT_REGISTRATION DATE NOT NULL,
    PRIMARY KEY (PATIENT_ID),
    FOREIGN KEY (PATIENT_CITY) REFERENCES CITY(CITY_ID),
    FOREIGN KEY (PATIENT_COUNTRY_ID) REFERENCES COUNTRY(COUNTRY_ID),
    FOREIGN KEY (PATIENT_SPORTS) REFERENCES SPORTS(SPORTS_ID))
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	,
    FOREIGN KEY(PATIENT_HISTORY_PERSONAL_ID) REFERENCES HISTORY_PERSONAL(PERSONAL_ID),
    FOREIGN KEY(PATIENT_HISTORY_FAMILY_ID) REFERENCES FAMILY(FAMILY_ID),
    FOREIGN KEY(PATIENT_PERSONAL_EXPERIENCE) REFERENCES EXPERIENCE(EXP_ID))
